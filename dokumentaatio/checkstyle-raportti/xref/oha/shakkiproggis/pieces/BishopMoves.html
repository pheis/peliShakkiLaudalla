6   package oha.shakkiproggis.pieces;
7   
8   import oha.shakkiproggis.PieceGroup;
9   import oha.shakkiproggis.pieces.Slideable;
10  import java.util.stream.Stream;
11  import oha.shakkiproggis.Square;
12  
13  /**
14   * Provides bishop moves for bishop and queen.
15   * @author pyry
16   */
17  public interface BishopMoves extends Slideable {
18  	/**
19  	 * Squares whre bishop attacks.
20  	 * @param s starting square
21  	 * @param my my pieces
22  	 * @param enemy enemy pieecs
23  	 * @return squares where bishop attacks.
24  	 */
25  	public default Stream<Square> attacks(Square s, PieceGroup my, PieceGroup enemy) {
26  		Stream<Square> leftD = slideOn((x, y)-> Square.onSameLeftDiagonal(x, y), s, my, enemy);
27  		Stream<Square> rightD = slideOn((x, y)-> Square.onSameRightDiagonal(x, y), s, my, enemy);
28  		return Stream.concat(leftD, rightD);
29  	}
30  }
