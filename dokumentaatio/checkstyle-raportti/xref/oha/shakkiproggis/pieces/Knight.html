6   package oha.shakkiproggis.pieces;
7   
8   import oha.shakkiproggis.PieceGroup;
9   import java.util.stream.Stream;
10  import oha.shakkiproggis.Square;
11  import oha.shakkiproggis.pieces.PieceObj;
12  
13  /**
14   * Knight.
15   * @author pyry
16   */
17  public class Knight implements PieceObj {
18  
19  	@Override
20  	public Stream<Square> attacks(Square s, PieceGroup friends, PieceGroup enemy) {
21  		return Square.allSquares()
22  			.filter((Square x) -> (Square.manhattanDistance(s, x) == 3))
23  			.filter((Square x) -> (Square.euclidianDistance(s, x) == 2));
24  	}	
25  }
