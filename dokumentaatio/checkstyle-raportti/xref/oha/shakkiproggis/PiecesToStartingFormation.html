17  import oha.shakkiproggis.pieces.PieceObj;

20  /**
21   * Provides methods to initialize a new start state for a chess game.
22   * Only needed by PieceGroup. This interface exists to make piecegroup's code more readable.
23   * @author pyry
24   */
25  public interface PiecesToStartingFormation {
26  	/**
27  	 * Generates the squares to pieces map.
28  	 * @param sqPtMap squares to piece enum map. Squares are piece locations.
29  	 * @param white if pieces are white true. if they are black then false.
30  	 */
31  	
32  	default void sqPtMapGen(EnumMap<Square, PieceT> sqPtMap, boolean white) {
33  		Square[] sqA = (white) ? Square.firstRank() : Square.eightRank();
34  		sqPtMap.put(sqA[0], PieceT.ROOK);
35  		sqPtMap.put(sqA[1], PieceT.KNIGHT);
36  		sqPtMap.put(sqA[2], PieceT.BISHOP);
37  		sqPtMap.put(sqA[3], PieceT.QUEEN);
38  		sqPtMap.put(sqA[4], PieceT.KING);
39  		sqPtMap.put(sqA[5], PieceT.BISHOP);
40  		sqPtMap.put(sqA[6], PieceT.KNIGHT);
41  		sqPtMap.put(sqA[7], PieceT.ROOK);
42  		if (white) {
43  			wPawnStartForm(sqPtMap);
44  		} else {
45  			bPawnStartForm(sqPtMap);
46  		}
47  	}
48  	/**
49  	 * Puts white pawns to their correct place on the begining of the game.
50  	 * @param sqPtMap Squares to Pieces enum map.  Squares are piece locations.
51  	 */
52  	default void wPawnStartForm(EnumMap<Square, PieceT> sqPtMap) {
53  		Square[] sqA = Square.secondRank();
54  		for (Square s : sqA) {
55  			sqPtMap.put(s, PieceT.WPAWN);
56  		}
57  	}
58  	/**
59  	 * Puts black pawn to starting formation on the befginning of the game.
60  	 * @param sqPtMap Squares to pieces enum map. Squares are piece locations. 
61  	 */
62  	default void bPawnStartForm(EnumMap<Square, PieceT> sqPtMap) {
63  		Square[] sqA = Square.seventhRank();
64  		for (Square s : sqA) {
65  			sqPtMap.put(s, PieceT.BPAWN);
66  		}
67  	}
68  	/**
69  	 * Generates the piece type to piece object map.
70  	 * @param pTpMap piece enum to piece object map.
71  	 * @param white true if pieces are white. else false.
72  	 */
73  	default void pTpMapGen(EnumMap<PieceT, PieceObj> pTpMap, boolean white) {
74  		pTpMap.put(PieceT.KING, new King());
75  		pTpMap.put(PieceT.QUEEN, new Queen());
76  		pTpMap.put(PieceT.ROOK, new Rook());
77  		pTpMap.put(PieceT.BISHOP, new Bishop());
78  		pTpMap.put(PieceT.KNIGHT, new Knight());
79  		if (white) {
80  			pTpMap.put(PieceT.WPAWN, new WhitePawn());
81  		} else {
82  			pTpMap.put(PieceT.BPAWN, new BlackPawn());
83  		}
84  	}
85  }
