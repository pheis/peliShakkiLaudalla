4   import java.util.*;
5   /**
6    * This represents a chess board. Has two piecesets on it. Enemys and mine. This is immutable object
7    * A new board is always generated when a move is done.
8    * During generations my pieces turn to enemy pieces and enemy pieces turn to my pieces.
9    * The 'me' of a Board if always the player whose turn it is.
10   * @author pyry
11   */
12  public class Board {
13  	public final PieceGroup my;
14  	public final PieceGroup enemy;
15  	/**
16  	 * Tells if curret player is white.
17  	 * if true current player is white.
18  	 * else current player is black.
19  	 */
20  
21  	public final boolean color;
22  	public final Move lastMove;
23  	/**
24  	 * This is the default constructor. 
25  	 * @param pc1 pawn promo target chooser for white player.
26  	 * @param pc2 pawn promo target chooser for black player.
27  	 */
28  	public Board(PawnPromoChooser pc1, PawnPromoChooser pc2) {
29  		this.my = new PieceGroup(true, pc1);
30  		this.enemy = new PieceGroup(false, pc2);
31  		this.color = true;
32  		this.lastMove = new Move(Square.A1, Square.A2);
33  	}
34  	/**
35  	 * This is the constructor that is used by the object itself to create more Board objects.
36  	 * @param my my pieces.
37  	 * @param enemy enemy pieces.
38  	 * @param color color of my.
39  	 * @param lMove last move that has been done on the game.
40  	 */
41  	public Board(PieceGroup my, PieceGroup enemy, boolean color, Move lMove) {
42  		this.my = my;
43  		this.enemy = enemy;
44  		this.color = color;
45  		this.lastMove = lMove;
46  	}
47  	// If enemy king is in check when it is my turn something illegal has happened. Else Everything OK.
48  	private boolean possibleFuture() {
49  		return !(enemy.kingInCheck(my));
50  	}
51  	/**
52  	 * Makes a new board.
53  	 * @param a starting square.
54  	 * @param b ending square.
55  	 * @return Maybe a new board. Maybe not. If the move was possible then yes, a new board.
56  	 */
57  	public Optional<Board> move(Square a, Square b) {
58  		PieceGroup futureEnemy = my.moveMyPiece(a, b);
59  		PieceGroup futureMe = enemy.eatOneOfMyPieces(a, b, this.my);
60  		Move futLstMove = new Move(a, b);
61  		Board futureBoard = new Board(futureMe, futureEnemy, !(this.color), futLstMove);
62  		return (futureBoard.possibleFuture()) ? Optional.of(futureBoard) : Optional.empty();
63  	}
64  	/**
65  	 * lists all possible moves.
66  	 * @return Arraylist of all possible gamestates after a legal move is done
67  	 */
68  	public ArrayList<Board> listPossibleMoves () {
69  		ArrayList<Board> boards = new ArrayList<>();
70  		my.squares()
71  			.flatMap(a -> my.moves(a, enemy).map(b -> move(a, b)))
72  			.filter(b -> b.isPresent())
73  			.map(b -> b.get())
74  			.forEach(b -> boards.add(b));
75  		return boards;
76  	}
77  	/**
78  	 * Tells how many pieces are still on the board.
79  	 * @return how many pieces are left.
80  	 */
81  	public int piecesOnBoard() {
82  		return my.sqPtMap.size() + enemy.sqPtMap.size();
83  	}
84  	
85  	
86  	/**
87  	 * Tells if current players King is in check.
88  	 * @return True or false.
89  	 */
90  	public boolean kingInCheck() {
91  		return my.kingInCheck(enemy);
92  	}
93  	/**
94  	 * Tells if two game positions are same.
95  	 * @param o object, other game position
96  	 * @return true if two game positions are the same
97  	 */
98  	@Override
99  	public boolean equals(Object o) {
100 		if (getClass() != o.getClass()) return false;
101 		Board other = (Board) o;
102 		//boolean t = (this.hashCode() == o.hashCode());
103 		boolean t = other.my.sqPtMap.equals(this.my.sqPtMap);
104 		t = t ? other.enemy.sqPtMap.equals(this.enemy.sqPtMap) : t;
105 		t = t ? (other.color == this.color) : t;
106 		if (t) {
107 			ArrayList<Long> possMoves = new ArrayList<>();
108 			other.listPossibleMoves().stream()
109 								.map(b -> b.lastMove.toLong())
110 								.forEach(l -> possMoves.add(l));
111 			for (Board b : listPossibleMoves()) {
112 				if (!possMoves.contains(b.lastMove.toLong())) return false;
113 			}
114 		}
115 		return t;
116 	}
117 	
118 	/**
119 	 * Tells about the last move.
120 	 * @return Last move that was done. 
121 	 */
122 	@Override
123 	public String toString() {
124 		return this.lastMove.toString();
125 	}
126 }
