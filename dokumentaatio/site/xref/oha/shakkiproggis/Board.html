110 		return Square.allSquares()
111 			.filter((Square x) -> (Square.manhattanDistance(s, x) == 3))
112 			.filter((Square x) -> (Square.euclidianDistance(s, x) == 2));
113 	}
114 	public Stream<Square> blackPawnAttacks(Square s) {
115 		return Square.allSquares()
116 			.filter((Square x) -> (Square.manhattanDistance(s, x) == 2))
117 			.filter((Square x) -> (Square.euclidianDistance(s, x) == 1))
118 			.filter((Square x) -> (x.ordinal() < s.ordinal()));
119 	}

132 	private Optional<Integer> infimum(BiPredicate<Square, Square> onSame, Square s) {
133 		return allSquaresWithAPiece()
134 			.filter((Square x) -> (onSame.test(x, s)))
135 			.filter((Square x) -> (x.ordinal() < s.ordinal()))
136 			.map((Square x) -> x.ordinal())
137 			.reduce((x, y) -> Math.max(x, y));
138 	}
139 	
140 	private Optional<Integer> supremum(BiPredicate<Square, Square> onSame, Square s) {
141 		return allSquaresWithAPiece()
142 			.filter((Square x) -> (onSame.test(x, s)))
143 			.filter((Square x) -> (x.ordinal() > s.ordinal()))
144 			.map((Square x) -> x.ordinal())
145 			.reduce((x, y) -> Math.min(x, y));
146 	}

148 	private Stream<Square> slideOn(BiPredicate<Square, Square> onSame, Square s) {
149 		return Square.allSquares()
150 			.filter((Square x) -> (onSame.test(s, x)))
151 			.filter((Square x) -> (x.ordinal() >= infimum(onSame, s).orElse(Square.A1.ordinal())))
152 			.filter((Square x) -> (x.ordinal() <= supremum(onSame, s).orElse(Square.H8.ordinal())));
153 	}

156 		Stream<Square> rank = slideOn((x, y)->Square.onSameRank(x, y), s);
157 		Stream<Square> file = slideOn((x, y)->Square.onSameFile(x, y), s);
158 		return Stream.concat(rank, file);
159 	}

161 	public Stream<Square> bishopAttacks(Square s) {
162 		Stream<Square> leftD = slideOn((x, y)->Square.onSameLeftDiagonal(x, y), s);
163 		Stream<Square> rightD = slideOn((x, y)->Square.onSameRightDiagonal(x, y), s);
164 		return Stream.concat(leftD, rightD);
165 	}

226 	public Stream<Square> whitePawnMoves(Square s) {
227 		//EnumSet<Square> moves = EnumSet.noneOf(Square.class);
228 		int moveDist = (s.ordinal() <= Square.H2.ordinal() && Square.A2.ordinal() <= s.ordinal()) ? 2 : 1;
229 		Stream<Square> moves = slideOn((x, y)->Square.onSameFile(x, y), s)
230 						.filter((Square x) -> (x.ordinal() > s.ordinal()))
231 						.filter((Square x) -> (Square.euclidianDistance(x, s) <= moveDist))
232 						.filter((Square x) -> !(mySquares().anyMatch(y -> y == x)))
233 						.filter(x -> !(enemySquares().anyMatch(y -> (x == y))));
234 						//.collect(Collectors.toCollection(() -> moves));
235 		Stream<Square> attackMoves = whitePawnAttacks(s)
236 						.filter(x -> (enemySquares().anyMatch(y -> y == x)));
237 						//.collect(Collectors.toCollection(() -> moves));
238 		return Stream.concat(moves, attackMoves);
239 	}

265 	public Stream<Square> movesOfThisPiece(Square s) {
266 		Piece p = this.myPieces.get(s);
267 		if (p == null) {
268 			return EnumSet.noneOf(Square.class).stream();
269 		}
270 		if (p == Piece.KING) {
271 			return kingMoves();
272 		}
273 		if (p == Piece.WPAWN) {
274 			return whitePawnMoves(s);
275 		}
276 		if (p == Piece.BPAWN) {
277 			return blackPawnMoves(s);
278 		} else {
279 			return linkPieceToAttack(s, p)
280 				.filter(x -> !(mySquares().anyMatch(y -> (x == y)))); // Remove squares occupied by me
281 				//.collect(Collectors.toCollection(() -> EnumSet.noneOf(Square.class)));
282 		}
283 	}

285 	private EnumMap<Square, Piece> moveMyPiece(Square a, Square b) {
286 		EnumMap<Square, Piece> myNewPieces = new EnumMap<>(Square.class);
287 		Piece movingPiece = this.myPieces.get(a);
288 		mySquares().filter(x -> x != a)
289 				   .forEach(x -> myNewPieces.put(x, this.myPieces.get(x)));
290 		myNewPieces.put(b, movingPiece);
291 		return myNewPieces;
292 	}

295 		EnumMap<Square, Piece> newEnemyPieces = new EnumMap<>(Square.class);
296 		enemySquares().filter(x -> !(x == b))
297 				   .forEach(x -> newEnemyPieces.put(x, this.enemyPieces.get(x)));
298 		return newEnemyPieces;
299 	}

302 		boolean t = (this.myPieces.get(a) == Piece.BPAWN || this.myPieces.get(a) == Piece.WPAWN);
303 		t = (t) ? (b.ordinal() - a.ordinal() == 16) : t;
304 		return t;

307 	private boolean[] generateBools(Square a) {
308 		boolean[] ts = new boolean[7];
309 		ts[0] = !(this.iAmWhite);
310 		ts[1] = (this.myKingMoved) ? true : (this.myPieces.get(a) != Piece.KING);
311 		ts[2] = (this.myARookMoved) ? true : (this.myPieces.get(a) != Piece.ROOK ? a == Square.A1 : false);
312 		ts[3] = (this.myARookMoved) ? true : (this.myPieces.get(a) != Piece.ROOK ? a == Square.H1 : false);
313 		ts[4] = this.enemyKingMoved;
314 		ts[5] = this.enemyARookMoved;
315 		ts[6] = this.enemyHRookMoved;
316 		return ts;
317 	}
