39  	public void printUI() {
40  		
41  		HashMap<Integer, String> myPieces = mv.getMyPieceStrings();		
42  		HashMap<Integer, String> enemyPieces = mv.getEnemyPieceStrings();	
43  		for (int j = 0; j < 8; j++) {
44  			for (int i = 56 - (8 * j); i <= 63 - (8 * j); i++) {		
45  				if (myPieces.containsKey(i)) {
46  					System.out.print(ANSI_BLUE + " " + myPieces.get(i).substring(0, 2) + " " + ANSI_RESET);
47  				} else if (enemyPieces.containsKey(i)) {
48  					System.out.print(ANSI_RED + " " + enemyPieces.get(i).substring(0, 2) + " " + ANSI_RESET);
49  				} else {
50  					System.out.print(" .. ");
51  				}
52  				//System.out.print(" "+this.ui[i]+" ");
53  			}
54  			System.out.printf("\n\n");
55  		}	
56  	}

58  	public void play() {
59  		Scanner sc = new Scanner(System.in);
60  		AI ai = new AI(mv);
61  		String input;
62  		boolean t;
63  		ai.move();
64  		while (true) {
65  			printUI();
66  			System.out.println("Anna liike");
67  			input = sc.nextLine();
68  			t = mv.movePiece(input);
69  			if (t) {
70  				ai.move();
71  			}
72  			System.out.println(t);	
73  		}
74  	}
