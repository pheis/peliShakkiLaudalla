26  	public static final String ANSI_RESET = "\u001B[0m";
27  	public static final String ANSI_BLACK = "\u001B[30m";
28  	public static final String ANSI_RED = "\u001B[31m";	
29  	public static final String ANSI_GREEN = "\u001B[32m";
30  	public static final String ANSI_YELLOW = "\u001B[33m";
31  	public static final String ANSI_BLUE = "\u001B[34m";
32  	public static final String ANSI_PURPLE = "\u001B[35m";
33  	public static final String ANSI_CYAN = "\u001B[36m";
34  	public static final String ANSI_WHITE = "\u001B[37m";
35  	/**
36  	 * constructor. yeps.
37  	 */
38  	public TextUI() {
39  		mv = new MoveValidator(pc1, pc2);
40  	}

43  	/**
44  	 * Prints some text.
45  	 */
46  	public void printUI() {
47  		
48  		HashMap<Integer, String> myPieces = mv.getMyPieceStrings();		
49  		HashMap<Integer, String> enemyPieces = mv.getEnemyPieceStrings();	
50  		for (int j = 0; j < 8; j++) {
51  			for (int i = 56 - (8 * j); i <= 63 - (8 * j); i++) {		
52  				if (myPieces.containsKey(i)) {
53  					System.out.print(ANSI_BLUE + " " + myPieces.get(i).substring(0, 2) + " " + ANSI_RESET);
54  				} else if (enemyPieces.containsKey(i)) {
55  					System.out.print(ANSI_RED + " " + enemyPieces.get(i).substring(0, 2) + " " + ANSI_RESET);
56  				} else {
57  					System.out.print(" .. ");
58  				}
59  				//System.out.print(" "+this.ui[i]+" ");
60  			}
61  			System.out.printf("\n\n");
62  		}	
63  	}
64  	/**
65  	 * lets play. 
66  	 */
67  	public void play() {
68  		Scanner sc = new Scanner(System.in);
69  		AI ai = new AI(mv, pc2);
70  		String input;
71  		boolean t;
72  		ai.move();
73  		while (true) {
74  			printUI();
75  			System.out.println("Anna liike");
76  			input = sc.nextLine();
77  			t = mv.move(input);
78  			if (t) {
79  				ai.move();
80  			}
81  			System.out.println(t);	
82  		}
83  	}
