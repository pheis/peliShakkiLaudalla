29  public class ChessGUI extends Application {
30  	int random = ThreadLocalRandom.current().nextInt(0, 2);	
31  	Iload il = new Iload();
32  	MoveValidator game = new MoveValidator();
33  	AI ai = new AI(game);
34  	int startX, startY, endX, endY;
35  	boolean doingMove = false;
36  	public static final int SQUARESIZE = 100;
37  	public static final int HEIGHT = 8;
38  	public static final int WIDTH = 8;
39  	private Group squares = new Group();
40  	private Group pieces = new Group();
41  	private ArrayList<ChessPiece> piecesList = new ArrayList<>();
42  	private Parent createContent() {
43  		Pane root = new Pane();
44  		root.setPrefSize(WIDTH * SQUARESIZE, HEIGHT * SQUARESIZE);	
45  		root.getChildren().addAll(squares, pieces);
46  		for (int i = 0; i < HEIGHT; i++) {
47  			for (int j = 0; j < WIDTH; j++) {
48  				ChessSquare sq = new ChessSquare((i + j) % 2 == 0, i, j);
49  				game.getMyPieces()
50  								.keySet()
51  								.forEach(p ->
52  								{
53  									int gX = coordToGuiCoordX(p);
54  									int gY = coordToGuiCoordY(p);
55  									ChessPiece piece;
56  									piece = new ChessPiece(gX, gY, game.getMyPieces().get(p), game.iAmWhite(), il);
57  									piecesList.add(piece);
58  									pieces.getChildren().add(piece);
59  								});
60  
61  				game.getEnemyPieces()
62  								.keySet()
63  								.forEach(p ->
64  								{		
65  									int gX = coordToGuiCoordX(p);
66  									int gY = coordToGuiCoordY(p);
67  									ChessPiece piece;
68  									piece = new ChessPiece(gX, gY, game.getEnemyPieces().get(p), !game.iAmWhite(), il);
69  									piecesList.add(piece);
70  									pieces.getChildren().add(piece);
71  								});
72  				
73  				squares.getChildren().add(sq);
74  			}
75  		}
76  		return root;
77  	}
78  	private void updCont() {
79  		piecesList.forEach(p -> this.pieces.getChildren().remove(p));
80  		piecesList.removeAll(piecesList);
81  		game.getMyPieces().keySet()
82  						.forEach(p ->
83  						{
84  							int gX = coordToGuiCoordX(p);
85  							int gY = coordToGuiCoordY(p);
86  							ChessPiece piece;
87  							piece = new ChessPiece(gX, gY, game.getMyPieces().get(p), game.iAmWhite(), il);
88  							piecesList.add(piece);
89  							pieces.getChildren().add(piece);
90  						});
91  				
92  				
93  		game.getEnemyPieces().keySet()
94  						.forEach(p ->
95  						{
96  							int gX = coordToGuiCoordX(p);
97  							int gY = coordToGuiCoordY(p);
98  							ChessPiece piece;
99  							piece = new ChessPiece(gX, gY, game.getEnemyPieces().get(p), !game.iAmWhite(), il);
100 							piecesList.add(piece);
101 							pieces.getChildren().add(piece);
102 						});
103 	}
104 	private int transformGuiCoordToBcoord(double x, double y) {
105 		int newX = (int) Math.floor(x / SQUARESIZE);
106 		int newY = 7 -  (int) Math.floor(y / SQUARESIZE);
107 		return newX + (8 * newY);
108 	}
109 	private int coordToGuiCoordX(int bCoord) {
110 		int x;
111 		x = bCoord % 8;
112 		//x *= 100;
113 		return x;
114 	}	
115 	private int coordToGuiCoordY(int bCoord) {
116 		// FIX THIS
117 		int y;
118 		y = 7 - (bCoord / 8);
119 		//y *= 100;
120 		return y;
121 	}
122 	@Override
123 	public void start(Stage primaryStage) throws Exception {	
124 		Scene scene = new Scene(createContent());	
125 		scene.setOnMousePressed((MouseEvent event) ->
126 		{
127 			if (doingMove == false) {
128 				startX = (int) Math.floor(event.getSceneX() / SQUARESIZE);
129 				startY = 7 - (int) Math.floor(event.getSceneY() / SQUARESIZE);
130 				doingMove = true;
131 			}			
132 		});		
133 		scene.setOnMouseReleased((MouseEvent event) ->
134 		{
135 			if (doingMove == true) {			
136 				endX = (int) Math.floor(event.getSceneX() / SQUARESIZE);
137 				endY = 7 - (int) Math.floor(event.getSceneY() / SQUARESIZE);			
138 				int startSq = startX + (8 * startY);
139 				int endSq = endX + (8 * endY);								
140 				if (0 <= startSq && startSq <= 63 && 0 <= endSq && endSq <= 63) {			
141 					StringBuilder sb = new StringBuilder();
142 					sb.append("");
143 					sb.append(Square.values()[startSq]);
144 					sb.append(Square.values()[endSq]);
145 					String wantedBoard = sb.toString();				
146 					boolean t;			
147 					t = game.movePiece(wantedBoard);				
148 					if (t) {
149 						updCont();				
150 						ai.move();	
151 						updCont();
152 					}	
153 					System.out.println(t);	
154 					System.out.println(wantedBoard);
155 				}	
156 			}
157 			doingMove = false;
158 		});	
159 		primaryStage.setTitle("Very High Quality Chess Program");	
160 		primaryStage.setScene(scene);
161 		primaryStage.show();
162 	}
